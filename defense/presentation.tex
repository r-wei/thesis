\frame{
	\frametitle{Background}
	%
	\only<1->
	{\begin{theorem}(Hochschild-Kostant-Rosenberg, '62) Let $A$ be a regular, commutative algebra 
	over a field $k$ of characteristic 0. Then, 
	\begin{align*}
	(C_\bullet(A, A), b) 
	& \xrightarrow{\sim}
	\Omega^\bullet_{A/k}\\
	(C^\bullet(A,A), \delta)
	& \xrightarrow{\sim}
	\wedge^\bullet T_A 
	= \wedge^\bullet(Der_k(A,A)).
	\end{align*}
	\end{theorem}}
	%
	\only<2>
	{\begin{theorem}(Kontsevich, '97) Let $A=C^\infty(M)$ 
	for $M$ a smooth real manifold. Then, there is an 
	$L_\infty$ map $$(C^{\bullet+1}(A,A), \delta, [,]_{Ger})
	\xrightarrow{\sim} (\wedge^{\bullet+1} T_A, d=0, [,]_{SN}).$$
	\end{theorem}}
	%
	\only<3>
	{\begin{theorem}(Tamarkin, '98) Dependent on the 
	choice of a Drinfeld associator, there is a 
	$Ger_\infty$ map $$(C^{\bullet+1}(A,A), \delta, 
	[,]_{Ger}, \cup, \dots)
	\xrightarrow{\sim} 
	(\wedge^\bullet T_A, d=0, [,]_{SN}, \wedge).$$
	\end{theorem}}
	%
	\only<4-5>
	{\begin{theorem}(Dolgushev-Tamarkin-Tsygan, '08) 
	There is a $Calc_\infty$ map 
	$$(C^\bullet(A,A), C_{-\bullet}(A,A))
	\xrightarrow{\sim} 
	(\wedge^\bullet T_A, \Omega^\bullet_{A/k}).$$
	\end{theorem}}
	%
	\only<5>
	{\begin{center}Can we use this calculus 
	structure to create a cyclic object?
	\end{center}}
}

\frame{
	\frametitle{A \textbf{sheafy-}cyclic object in DGCocat}
	\begin{defn} Let $\chi$ be the category with 
	objects $\{A_0 \to A_1 \to \dots \to A_n \to A_0\}$ 
	and morphisms compositions of 
	{\small  
  	\setlength{\abovedisplayskip}{6pt}
  	\setlength{\belowdisplayskip}{\abovedisplayskip}
  	\setlength{\abovedisplayshortskip}{0pt}
  	\setlength{\belowdisplayshortskip}{3pt}
  	\begin{align*}
	rotations\; \tau_n
	&: 
	\mathcal{A}
	\mapsto (A_n \to A_0 \to \dots \to A_n) \\
	coboundaries\; \delta_{j,n}
	&: 
	\mathcal{A}\mapsto 
	(A_0 \to \dots \to A_j \to A_{j+2\,(mod\,n+1)} 
	\to \dots \to A_0) \\
	codegeneracies\; \sigma_{i,n}
	&: \mathcal{A}
	\mapsto 
	(A_0 \to \dots \to A_i \to A_i \to \dots \to A_0)
	\end{align*}}
	where $\mathcal{A}:= (A_0 \to \dots \to A_n \to A_0)$,
	subject to the cyclic relations.
	\end{defn}

	\only<2->
	{\begin{prop}We have a functor
	$\chi \to DGCocat$
	\begin{align*}
	Objects:\; (A_0 \to \dots \to A_n \to A_0)
	&\mapsto 
	B(A_0 \to \dots \to A_n \to A_0)\\
	Generating\; morphisms:\; \lambda
	&\mapsto 
	\hat{\lambda}
	\end{align*}
	\end{prop}}
}

\frame{
	\frametitle{Dg comodules over dg cocategories}

	\begin{defn}
	A \textbf{dg comodule} $C$ 
	over a dg cocategory $B$ consists
	of the following data:
	\begin{itemize}
	\item for each object $f \in B$, a complex $C^\bullet(f)$, and
	\item maps of complexes
	\begin{equation*}
	\Delta_C(f): C^\bullet(f) \to 
	\prod\limits_{g \in Obj(B)} B^\bullet(f,g) \otimes C^\bullet(g).
	\end{equation*}
	\end{itemize}
	{\small such that the following diagrams 
	for coassociativity and 
	counitality commute:}
	$$
	\resizebox{.9\vsize}{!}
	{\xymatrixcolsep{3pc}
	\xymatrixrowsep{3pc}
	\xymatrix{
	C^\bullet(f)
	\ar[r]^{\Delta_C}
	\ar[d]_{\Delta_C}
	& \prod \limits_{g \in Obj(B)}
	B^\bullet(f,g) \otimes
	C^\bullet(g)
	\ar[d]^{ id_B \otimes \Delta_C}\\
	\prod \limits_{g \in Obj(B)}
	B^\bullet(f,g) \otimes
	C^\bullet(g)
	\ar[r]^{\Delta_B \otimes id_C}
	& \prod \limits_{\substack{g, g^\prime \in\\ Obj(B)}}
	B^\bullet(f,g) \otimes
	B^\bullet(g,g^\prime) \otimes
	C^\bullet(g^\prime)  
	}}
	\resizebox{.5\vsize}{!}
	{\xymatrixcolsep{3pc}
	\xymatrixrowsep{3pc}
	\xymatrix{
	C^\bullet(f)
	\ar[r]^{\Delta_C}
	\ar[rd]_{id}
	& \prod \limits_{g \in Obj(B)}
	B^\bullet(f,g) \otimes
	C^\bullet(g)
	\ar[d]^{\epsilon_B \otimes id_C} \\
	& C^\bullet(f)
	}}
	$$
	\end{defn}
}
